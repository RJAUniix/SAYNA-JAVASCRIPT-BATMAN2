var sliderContainer = document.querySelector('.slider-container');
var images = document.querySelectorAll('.slider-container img');
var imageIndex = 0;
var imageWidth = images[0].clientWidth;
var intervalId;

// Fonction pour afficher l'image suivante
function nextImage() {
  imageIndex++;
  if (imageIndex === images.length) {
    imageIndex = 0;
  }
  updateSlider();
}

// Fonction pour mettre à jour la position du slider
function updateSlider() {
  sliderContainer.style.transform = 'translateX(' + (-imageIndex * imageWidth) + 'px)';
}

// Fonction pour démarrer le slider automatique
function startSlider() {
  intervalId = setInterval(nextImage, 5000); // Change d'image toutes les 5 secondes (5000 millisecondes)
}

// Fonction pour arrêter le slider automatique
function stopSlider() {
  clearInterval(intervalId);
}

// Écouteur d'événement pour redémarrer le slider automatique lorsque la souris est hors du slider
sliderContainer.addEventListener('mouseleave', startSlider);

// Appeler la fonction startSlider() pour démarrer le slider automatique au chargement initial de la page
startSlider();

// controle de la souris

var canvas = document.getElementById('canvas');
var ctx = canvas.getContext('2d');
var raf;
var running = false;

var ball = {
  x: 100,
  y: 100,
  vx: 5,
  vy: 1,
  radius: 25,
  color: 'blue',
  draw: function() {
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.radius, 0, Math.PI*2, true);
    ctx.closePath();
    ctx.fillStyle = this.color;
    ctx.fill();
  }
};

function clear() {
  ctx.fillStyle = 'rgba(255,255,255,0.3)';
  ctx.fillRect(0,0,canvas.width,canvas.height);
}

function draw() {
  clear();
  ball.draw();
  ball.x += ball.vx;
  ball.y += ball.vy;

  if (ball.y + ball.vy > canvas.height || ball.y + ball.vy < 0) {
    ball.vy = -ball.vy;
  }
  if (ball.x + ball.vx > canvas.width || ball.x + ball.vx < 0) {
    ball.vx = -ball.vx;
  }

  raf = window.requestAnimationFrame(draw);
}

canvas.addEventListener('mousemove', function(e){
  if (!running) {
    clear();
    ball.x = e.clientX;
    ball.y = e.clientY;
    ball.draw();
  }
});

canvas.addEventListener("click",function(e){
  if (!running) {
    raf = window.requestAnimationFrame(draw);
    running = true;
  }
});

canvas.addEventListener("mouseout",function(e){
  window.cancelAnimationFrame(raf);
  running = false;
});

ball.draw();
